\section{GMP with \cpp}

\subsection{Overview}

\begin{frame}
\frametitle{Overview}
\begin{itemize}
\item GMP is natively written in \c
\item most \c{} constructs can be used in \cpp
\pause \item GMP \cpp{} interface is convenient:
  \begin{itemize}
  \item templates
  \item classes
  \item overloaded functions/operators
  \end{itemize}
\end{itemize}
\end{frame}
