% generateKeys
\codes{\begin{tikzpicture}
  \begin{umlseqdiag}
    \umlactor{Receiver}
    \umlobject{Cry}
    \umlobject{GitHub}
    \begin{umlcall}
    [op={generate keys}, return={create public and private
      keys}, padding=5]{Receiver}{Cry}
    \end{umlcall}
    \begin{umlcall}[op={push public key}]{Receiver}{GitHub}
    \end{umlcall}
  \end{umlseqdiag}
\end{tikzpicture}}

% encrypt
\codes{\begin{tikzpicture}
  \begin{umlseqdiag}
    \umlactor{Sender}
    \umlobject{Cry}
    \umlobject{GitHub}
    \begin{umlcall}[op={get public key}]{GitHub}{Sender}
    \end{umlcall}
    \begin{umlcall}[op={encrypt plaintext}, return={create
      ciphertext}, padding=5]{Sender}{Cry}
    \end{umlcall}
    \begin{umlcall}[op={push ciphertext}]{Sender}{GitHub}
    \end{umlcall}
  \end{umlseqdiag}
\end{tikzpicture}}

% decrypt
\codes{\begin{tikzpicture}
  \begin{umlseqdiag}
    \umlactor{Receiver}
    \umlobject{Cry}
    \umlobject{GitHub}
    \begin{umlcall}[op={get ciphertext}]{GitHub}{Receiver}
    \end{umlcall}
    \begin{umlcall}[op={decrypt ciphertext}, return={create
      plaintext}, padding=5]{Receiver}{Cry}
    \end{umlcall}
  \end{umlseqdiag}
\end{tikzpicture}}

% cryptanalyze
\codes{\begin{tikzpicture}
  \begin{umlseqdiag}
    \umlactor{Cryptographer}
    \umlobject{Cry}
    \umlobject{GitHub}
    \begin{umlcall}[op={get public key and ciphertext}]
      {GitHub}{Cryptographer}
    \end{umlcall}
    \begin{umlcall}[op={run cryptanalysis}, return={display
      results}, padding=5]{Cryptographer}{Cry}
    \end{umlcall}
  \end{umlseqdiag}
\end{tikzpicture}}
