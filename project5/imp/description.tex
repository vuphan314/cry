\subsection{Implementation Description}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Dependencies
\begin{itemize}
  \item GMP
\end{itemize}

Compilation
\begin{itemize}
\item \code{make -C cry/src}
\end{itemize}

Use
\begin{itemize}
\item Help
  \begin{itemize}
  \item \code{./cry}
\end{itemize}
\item Testing
  \begin{itemize}
  \item \code{./cry t <spaceless-plaintext> <\cs>}
  \end{itemize}
\item Key-Generation
  \begin{itemize}
  \item \code{./cry g <receiver> <\cs>}
  \end{itemize}
\item Encryption
  \begin{itemize}
  \item \code{./cry e <sender> <receiver>}
  \end{itemize}
\item Decryption
  \begin{itemize}
  \item \code{./cry d <receiver> <sender>}
  \end{itemize}
\item Cryptanalysis
  \begin{itemize}
  \item \code{./cry c <receiver> <sender>}
  \end{itemize}
\end{itemize}
